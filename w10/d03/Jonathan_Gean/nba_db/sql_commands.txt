Gutenberg

1) SELECT author_id, count(title) FROM books GROUP BY author_id ORDER BY count(title) DESC LIMIT 20;
2) SELECT LEFT(name, 1), count(id) FROM authors GROUP BY LEFT(name, 1);
3) SELECT name, count(b.author_id) FROM books AS b INNER JOIN authors AS a ON b.author_id=a.id GROUP BY a.name ORDER BY count(b.author_id) DESC LIMIT 20;

Bonus:
1) SELECT a.name, b.title FROM books AS b INNER JOIN authors AS a ON b.author_id = a.id WHERE b.title LIKE '%Love%' ORDER BY a.name LIMIT 15;
