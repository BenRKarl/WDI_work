1. Create a database
====================

CREATE DATABASE cowcow_farm;

2. Create tables
================

CREATE TABLE farmers(id SERIAL PRIMARY KEY, name varchar(100));
CREATE TABLE cows(id SERIAL PRIMARY KEY, farmer_id integer references farmers(id), name varchar(100));

4. select statements
====================

4.1. SELECT name FROM cows WHERE name LIKE '%ooo%';

4.2. SELECT f.name, c.name FROM farmers AS f INNER JOIN cows AS c ON f.id = c.farmer_id;

4.3. SELECT name FROM farmers WHERE (SELECT count(*) FROM cows WHERE farmer_id = farmers.id) > 1;
